```xml
<!--在添加时，根据当前登录信息，分页模糊查询,用户编号位于:AGI.AUR.USR_NO-->
<select id="queryAuthGroupWhenAdd" resultMap="BaseResultMap">
    SELECT * FROM (
    SELECT
    *
    FROM
    AUTH_GROUP_INFO AGI
    WHERE AGI.GROUP_ID IN
    (SELECT DISTINCT AUR.GROUP_ID
    FROM AUTH_UGROUP_REL AUR
    WHERE AUR.USR_NO = #{usrNo})
    ) ANS
    <where>
        <if test="authGroupInfo.groupNm != null and authGroupInfo.groupNm != '' ">
            <bind name="patternGroupNm" value="'%'+authGroupInfo.groupNm+'%'"/>
            and ANS.GROUP_NM like #{patternGroupNm}
        </if>
        <if test="authGroupInfo.groupId != null and authGroupInfo.groupId != '' ">
            <bind name="patternGroupId" value="'%'+authGroupInfo.groupId+'%'"/>
            and ANS.GROUP_ID like #{patternGroupId}
        </if>
        <if test="authGroupInfo.groupTp != null and authGroupInfo.groupTp != '' ">
            <bind name="patternGroupTp" value="'%'+authGroupInfo.groupTp+'%'"/>
            and ANS.GROUP_TP like #{patternGroupTp}
        </if>
        <if test="authGroupInfo.dataClass != null and authGroupInfo.dataClass != '' ">
            and ANS.DATA_CLASS = #{authGroupInfo.dataClass}
        </if>
    </where>
    ORDER BY ANS.CREATE_TIME desc
</select>
```
